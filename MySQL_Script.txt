SELECT 
    CASE 
        WHEN price_usd < 400 THEN 'Budget'
        WHEN price_usd BETWEEN 400 AND 800 THEN 'Mid-range'
        ELSE 'Flagship'
    END AS price_segment,
    COUNT(*) AS num_models,
    ROUND(AVG(rating), 2) AS avg_rating
FROM mobile_phones_2025
GROUP BY price_segment
ORDER BY num_models DESC;
